from js import document, window from pyodide.ffi import create_proxy output_div = document.getElementById("output") user_input = document.getElementById("user-input") def execute_python(event): if event.key == "Enter": code = user_input.value if not code.strip(): return # Display the command new_command = document.createElement("div") new_command.innerHTML = f'>>> {code}' output_div.appendChild(new_command) # Execute and catch results try: # eval() for single expressions, exec() for statements # We'll try to evaluate it first to get a return value result = eval(code) if result is not None: res_div = document.createElement("div") res_div.style.color = "#94a3b8" res_div.innerText = f" {result}" output_div.appendChild(res_div) except Exception as e: try: exec(code) except Exception as e2: err_div = document.createElement("div") err_div.style.color = "#ef4444" err_div.innerText = f" Error: {e2}" output_div.appendChild(err_div) # Reset and scroll user_input.value = "" output_div.scrollTop = output_div.scrollHeight # Create a proxy for the event listener on_keydown = create_proxy(execute_python) user_input.addEventListener("keydown", on_keydown)
